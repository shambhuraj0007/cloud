Then go to setting then crete apex class and paste:

public class CalculatorApp {

    // Addition
    public static Integer addNumbers(Integer a, Integer b) {

        Integer result = a + b;

        System.debug('Addition = ' + result);

        return result;
    }

    // Subtraction
    public static Integer subtractNumbers(Integer a, Integer b) {

        Integer result = a - b;

        System.debug('Subtraction = ' + result);

        return result;
    }

    // Multiplication
    public static Integer multiplyNumbers(Integer a, Integer b) {

        Integer result = a * b;

        System.debug('Multiplication = ' + result);

        return result;
    }

    // Division
    public static Decimal divideNumbers(Integer a, Integer b) {

        if(b == 0) {

            System.debug('Cannot divide by zero');

            return 0;
        }

        Decimal result = (Decimal)a / b;

        System.debug('Division = ' + result);

        return result;
    }
}
