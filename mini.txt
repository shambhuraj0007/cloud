Then go to setting then crete apex class and paste:

public class StudentSystemController {

    public String studentName {get; set;}
    public String course {get; set;}

    public void saveStudent() {

        Account acc = new Account();

        acc.Name = studentName + ' - ' + course;

        insert acc;

        ApexPages.addMessage(
            new ApexPages.Message(
                ApexPages.Severity.CONFIRM,
                'Student Record Saved Successfully'
            )
        );

        studentName = '';
        course = '';
    }
}

Create new visualforce page:
maincode

<apex:page controller="StudentSystemController">

    <apex:form>

        <apex:pageBlock title="Student Management System">

            <apex:pageMessages />

            <apex:pageBlockSection columns="1">

                <apex:inputText value="{!studentName}"
                                label="Student Name"/>

                <apex:inputText value="{!course}"
                                label="Course"/>

            </apex:pageBlockSection>

            <apex:commandButton value="Save Student"
                                action="{!saveStudent}"/>

        </apex:pageBlock>

    </apex:form>

</apex:page>
